6   package headsuphud.main;
7   
8   
9   import headsuphud.handdata.DataStorage;
10  import headsuphud.handreader.HandHistoryTextToObjectsConverter;
11  import headsuphud.handreader.HandReader;
12  import headsuphud.gui.HudGUI;
13  
14  /**
15   *
16   * @author Juuso
17   */
18  public class HeadsupHUD {
19  
20      public static void main(String[] args) {
21  
22          DataStorage datastorage = loadStatistics();
23          HudGUI jframe = new HudGUI(datastorage);
24          jframe.run(datastorage);
25  
26      }
27  
28      /**
29       * Metodi lataa analysoitavat kädet ja palauttaa datastoragen, jotta
30       * käsiin pääsee kiinni myöhemmin.
31       */
32      public static DataStorage loadStatistics() {
33          HandReader handreader = new HandReader();
34          handreader.loadFileContents("Test.txt");
35  
36          HandHistoryTextToObjectsConverter converter = new HandHistoryTextToObjectsConverter(handreader.getHandData());
37          converter.convert();
38  
39          return converter.getDatastorage(); 
40      }
41  }
