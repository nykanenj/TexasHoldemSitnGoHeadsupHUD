5   package headsuphud.handdata;

7   import static headsuphud.handdata.Action.*;
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  public class ActionStatsOnStreet {
12  
13      private final Street street;
14      private List<Action> actions;
15      private int count;
16  
17      public ActionStatsOnStreet(Street street) {
18          this.street = street;
19          actions = new ArrayList<>();
20      }
21  
22      /**
23       * Metodi lisaa tallennettavan komennon komentoja sisaltavaan listaan.
24       *
25       * @param action Tallennettava komento
26       *
27       */
28      public void addAction(Action action) {
29          if (action == Fold || action == Check || action == Raise) {  //PROBLEM: We should count calls when someone donk bets us! 
30              count++;
31          }
32          actions.add(action);
33      }
34  
35      public List<Action> getActions() {
36          return actions;
37      }
38  
39      /**
40       * Metodi antaa karkeaa statistiikkaa miten usein tiettyä komentoa on tehty
41       *
42       * @param action Komento, josta halutaan tietää statistiikkaa
43       *
44       * @return Miten usein pelaaja on käyttänyt tätä komentoa suhteessa muihin
45       * komentoihin
46       */
47      public double stats(Action action) {
48          if (count == 0) {
49              return -1;
50          }
51  
52          int occurrences = 0;
53          for (Action actioninstance : actions) {
54              if (actioninstance == action) {
55                  occurrences++;
56              }
57          }
58          return occurrences / count;
59      }

61  }
