5   package headsuphud.handdata;

7   public class Hand {
8   
9       private final String handid;
10      private final Player player1;
11      private final Player player2;
12      private Card[] flopTurnRiverCards;
13  
14      /**
15       * Konstruktori luokalle Hand.
16       *
17       * @param handid pelatun käden id.
18       * @param flopTurnRiverCards Taulukko, joka sisältää pöytään tulevat kortit.
19       */
20      public Hand(String handid, Player player1, Player player2, Card[] flopTurnRiverCards) {
21          this.handid = handid;
22          this.player1 = player1;
23          this.player2 = player2;
24          this.flopTurnRiverCards = flopTurnRiverCards;
25      }
26  
27      public String getHandid() {
28          return handid;
29      }
30  
31      public Player[] getPlayers() {
32          Player[] players = new Player[2];
33          players[0] = player1;
34          players[1] = player2;
35          return players;
36      }
37  
38      public Card[] getFlopTurnRiverCards() {
39          return flopTurnRiverCards;
40      }

42  }
